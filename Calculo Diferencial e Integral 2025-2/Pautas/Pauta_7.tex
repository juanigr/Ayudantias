\begin{defi}[Puntos crรญticos de una funciรณn] Sea $f : A \subseteq \mathbb{R} \to \mathbb{R}$ una funciรณn, donde $A$ es su dominio. Sea $\bar{x} \in A$.
\begin{itemize}
    \item Diremos que $\bar{x}$ es un punto de \textbf{mรญnimo global} de $f$, si $f(\bar{x}) \le f(x)$ para todo $x \in A$.
    De manera anรกloga se define un punto de \textbf{mรกximo global} de $f$.
    
    \item Diremos que $\bar{x}$ es un punto de \textbf{mรญnimo local} de $f$ si $\exists \delta > 0$ tal que
    \[
    f(\bar{x}) \le f(x) \quad \forall x \in A \cap [\bar{x} - \delta, \bar{x} + \delta].
    \]
    De manera anรกloga se define un punto de \textbf{mรกximo local} de $f$.
\end{itemize}
\end{defi}
Si se cumple cualquiera de los 4 casos, diremos que $\bar{x}$ es un punto crรญtico o extremo de la funciรณn $f$.

\begin{teo}
Si $\bar{x} \in A$ es un punto crรญtico de una funciรณn $f : A \subseteq \mathbb{R} \to \mathbb{R}$, y $f$ es derivable en $\bar{x}$, entonces $f'(\bar{x}) = 0$.
\end{teo}

\begin{teo}
Si $f$ es diferenciable en $a$, entonces $f$ es continua en $a$.
\end{teo}
\begin{teo}[Teorema del Valor Extremo (de Weierstrass)]
Si $f$ es \textbf{continua} en $[a, b]$, entonces $f$ alcanza mรกximo y mรญnimo absolutos en $[a, b]$.\end{teo}

\begin{teo}[Regla de lโHรดpital] Sean $f, g : (a, b) \to \mathbb{R}$ derivables en $(a, b)$, tales que
\[
\lim_{x \to a^+} f(x) = \lim_{x \to a^+} g(x) = L
\]
con $L = 0$ o $L = \pm\infty$, y $g'(x) \neq 0$ para todo $x \in (a, b)$. Entonces
\[
\lim_{x \to a^+} \frac{f(x)}{g(x)} = \lim_{x \to a^+} \frac{f'(x)}{g'(x)}
\]
siempre que este รบltimo lรญmite exista.
\end{teo}

\begin{teo} Sea $f : [a, b] \to \mathbb{R}$ continua en $[a, b]$ y derivable en $(a, b)$. Si $f'(x) \ge 0$ (resp. $\le 0$) para todo $x \in (a, b)$, entonces $f$ es creciente (resp. decreciente) en $[a, b]$. Si la desigualdad es estricta, la monotonรญa es igualmente estricta.
\end{teo}

\begin{prop}
    Sea $f : (a, b) \to \mathbb{R}$, $k$ veces derivable en $\bar{x} \in (a, b)$, con $f'(\bar{x}) = \dots = f^{[k-1]}(\bar{x}) = 0$ y $f^{[k]}(\bar{x}) \neq 0, k \ge 2$. Entonces hay 3 casos posibles:
\begin{enumerate}
    \item Si $k$ es par y $f^{[k]}(\bar{x}) > 0$, $\bar{x}$ es un \textbf{mรญnimo local}.
    \item Si $k$ es par y $f^{[k]}(\bar{x}) < 0$, $\bar{x}$ es un \textbf{mรกximo local}.
    \item Si $k$ es impar, $\bar{x}$ es un \textbf{punto de inflexiรณn}.
\end{enumerate}
\end{prop}
